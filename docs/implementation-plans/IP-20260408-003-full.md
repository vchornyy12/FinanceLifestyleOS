# Implementation Plan: Phase 3 — Banking Integration

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Connect PKO BP (and other Polish banks) via GoCardless Bank Account Data API so transactions are auto-imported, displayed alongside manual/OCR entries, reconciled with receipts, and pushed to the user's phone.

**Architecture:** Next.js API routes handle all GoCardless OAuth and sync logic (server-side, Anthropic-key-style billing isolation). Mobile connects via deep link to the web OAuth flow; Expo Push Notifications deliver sync alerts. A polling sync endpoint is called on-demand (sync button) and can be wired to any scheduler.

**Tech Stack:** GoCardless Bank Account Data API v2 (formerly Nordigen) · Next.js App Router API routes · Supabase (Postgres + RLS) · Expo Push Notifications · Zod · pnpm monorepo

---

## Key Technical Decisions

| Decision | Rationale |
|----------|-----------|
| GoCardless Bank Account Data (not Salt Edge) for Phase 3 | Free tier, better docs, Salt Edge as fallback in Phase 4 |
| Polling sync (not webhooks) | Bank Account Data free tier has no webhooks; on-demand + scheduler covers the 24h SLA |
| Mobile connects via `Linking.openURL` to web | GoCardless OAuth is a browser flow; avoids WebView session issues |
| Expo Push via Expo's push service | No FCM/APNs config needed for EAS builds using Expo's gateway |
| `reference` = random UUID (not user_id) | Prevents user enumeration on the OAuth callback |
| Reconciliation by amount + date window | Simple and correct for Polish retail; fuzzy merchant matching is Phase 4 |

## Risk Factors

| Risk | Mitigation |
|------|------------|
| GoCardless requires registration + approval | Create account at bankaccountdata.gocardless.com before T2. Free sandbox available immediately |
| PKO BP OAuth may require eKYC / business verification | Test with a supported sandbox bank first (GoCardless provides sandbox IBANs) |
| GoCardless access token expires in 24h | Client-credentials token fetched fresh on each sync call — no refresh logic needed |
| Push token invalid / user revoked permission | Wrap Expo push send in try/catch; log silently; never block sync on push failure |
| Duplicate transaction import on re-sync | `external_id` unique index + `ON CONFLICT DO NOTHING` prevents duplicates |

## Codebase Conventions (existing — follow these)

| Convention | Pattern |
|------------|---------|
| Server action errors | `{ fieldErrors }` for validation, `{ error: '...' }` for failures — never raw Supabase error messages |
| Server action auth | Always `supabase.auth.getUser()` — never trust client user_id |

| API route Supabase client | Re-init from `NextRequest` cookies — cannot use `next/headers`-based server.ts |

```typescript
// apps/web/lib/banking/goCardlessClient.ts

const BASE_URL = 'https://bankaccountdata.gocardless.com/api/v2'

export interface GCInstitution {
  id: string
  name: string
  logo: string
  countries: string[]
}

export interface GCRequisition {
  id: string
  link: string
  status: string
  accounts: string[]
  reference: string
}

export interface GCTransaction {
  externalId: string        // transactionId or internalTransactionId
  bookingDate: string       // YYYY-MM-DD
  amount: string            // e.g. "-42.50"
  currency: string          // e.g. "PLN"
  merchant: string          // creditorName or remittanceInfo fallback
}

// Fetch a fresh client-credentials access token.
// GoCardless tokens last 24h; we fetch fresh per sync to avoid refresh logic.
export async function getAccessToken(): Promise<string> {
  const res = await fetch(`${BASE_URL}/token/new/`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      secret_id: process.env.GOCARDLESS_SECRET_ID,
      secret_key: process.env.GOCARDLESS_SECRET_KEY,
    }),
  })
  if (!res.ok) throw new Error(`GoCardless token fetch failed: ${res.status}`)
  const data = await res.json()
  return data.access as string
}

export async function listInstitutions(country: string): Promise<GCInstitution[]> {
  const token = await getAccessToken()
  const res = await fetch(`${BASE_URL}/institutions/?country=${country}`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) throw new Error(`GoCardless institutions fetch failed: ${res.status}`)
  return res.json() as Promise<GCInstitution[]>
}

export async function createRequisition(opts: {
  institutionId: string
  redirectUrl: string
  reference: string
}): Promise<{ id: string; link: string }> {
  const token = await getAccessToken()
  const res = await fetch(`${BASE_URL}/requisitions/`, {
    method: 'POST',
    headers: {
      Authorization: `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      redirect: opts.redirectUrl,
      institution_id: opts.institutionId,
      reference: opts.reference,
      user_language: 'PL',
    }),
  })
  if (!res.ok) throw new Error(`GoCardless requisition create failed: ${res.status}`)
  const data = await res.json()
  return { id: data.id as string, link: data.link as string }
}

export async function getRequisition(requisitionId: string): Promise<GCRequisition> {
  const token = await getAccessToken()
  const res = await fetch(`${BASE_URL}/requisitions/${requisitionId}/`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) throw new Error(`GoCardless requisition fetch failed: ${res.status}`)
  return res.json() as Promise<GCRequisition>
}

export async function getAccountTransactions(
  accountId: string,
  dateFrom?: string,
): Promise<GCTransaction[]> {
  const token = await getAccessToken()
  const params = dateFrom ? `?date_from=${dateFrom}` : ''
  const res = await fetch(`${BASE_URL}/accounts/${accountId}/transactions/${params}`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) throw new Error(`GoCardless transactions fetch failed: ${res.status}`)
  const data = await res.json()
  const booked: Record<string, unknown>[] = data?.transactions?.booked ?? []
  return booked.map((t) => ({
    externalId: (t.transactionId ?? t.internalTransactionId ?? '') as string,
    bookingDate: t.bookingDate as string,
    amount: (t.transactionAmount as { amount: string }).amount,
    currency: (t.transactionAmount as { currency: string }).currency,
    merchant:
      ((t.creditorName ?? t.debtorName ?? t.remittanceInformationUnstructured ?? 'Unknown') as string),
  }))
}
```

```typescript
'use server'

import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'
import { createClient } from '@/lib/supabase/server'
import { createRequisition, listInstitutions, GCInstitution } from '@/lib/banking/goCardlessClient'
import { BankConnectionInsert } from '@/types/database'

const SITE_URL = process.env.NEXT_PUBLIC_SITE_URL ?? 'http://localhost:3000'

export type BankingActionState = { error?: string; success?: boolean } | null

// Returns the list of Polish institutions for the connect dialog.
export async function getPolishInstitutions(): Promise<GCInstitution[]> {
  return listInstitutions('PL')
}

// Initiates GoCardless OAuth: creates a requisition, stores it as pending,
// then redirects the user to the GoCardless bank selection / auth page.
export async function initiateConnection(institutionId: string, institutionName: string) {
  const supabase = await createClient()
  const { data: { user }, error: userError } = await supabase.auth.getUser()
  if (userError || !user) redirect('/login')

  const referenceToken = crypto.randomUUID()
  const redirectUrl = `${SITE_URL}/api/banking/callback`

  const { id: requisitionId, link } = await createRequisition({
    institutionId,
    redirectUrl,
    reference: referenceToken,
  })

  const connection: BankConnectionInsert = {
    user_id: user.id,
    institution_id: institutionId,
    institution_name: institutionName,
    requisition_id: requisitionId,
    reference_token: referenceToken,
    status: 'pending',
  }

  const { error } = await supabase.from('bank_connections').insert(connection)
  if (error) return { error: 'Failed to save bank connection.' }

  redirect(link)
}

// Disconnects a bank connection and removes its synced transactions.
export async function disconnectBank(
  _prevState: BankingActionState,
  formData: FormData,
): Promise<BankingActionState> {
  const connectionId = formData.get('connectionId') as string | null
  if (!connectionId) return { error: 'Invalid connection ID.' }

  const supabase = await createClient()
  const { data: { user }, error: userError } = await supabase.auth.getUser()
  if (userError || !user) return { error: 'Not authenticated.' }

  // Null out bank_connection_id on transactions (keep the transactions)
  await supabase
    .from('transactions')
    .update({ bank_connection_id: null })
    .eq('bank_connection_id', connectionId)
    .eq('user_id', user.id)

  const { error } = await supabase
    .from('bank_connections')
    .delete()
    .eq('id', connectionId)
    .eq('user_id', user.id)

  if (error) return { error: 'Failed to disconnect bank.' }

  revalidatePath('/dashboard/banking')
  return { success: true }
}
```

- [ ] **Step 2: Create the OAuth callback route**

```typescript
// apps/web/app/api/banking/callback/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { createRouteHandlerClient } from '@supabase/auth-helpers-nextjs'
import { cookies } from 'next/headers'
import { getRequisition } from '@/lib/banking/goCardlessClient'

const SITE_URL = process.env.NEXT_PUBLIC_SITE_URL ?? 'http://localhost:3000'

export async function GET(req: NextRequest) {
  // GoCardless sends ?ref=<reference_token> on successful OAuth
  const referenceToken = req.nextUrl.searchParams.get('ref')
  if (!referenceToken) {
    return NextResponse.redirect(`${SITE_URL}/dashboard/banking?error=missing_ref`)
  }

  // Use service role to find the pending connection by reference_token
  const { createClient: createAdminClient } = await import('@supabase/supabase-js')
  const adminSupabase = createAdminClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { auth: { autoRefreshToken: false, persistSession: false } }
  )

  const { data: connection, error: findError } = await adminSupabase
    .from('bank_connections')
    .select('id, requisition_id')
    .eq('reference_token', referenceToken)
    .eq('status', 'pending')
    .single()

  if (findError || !connection) {
    console.error('[banking/callback] connection not found for ref:', referenceToken)
    return NextResponse.redirect(`${SITE_URL}/dashboard/banking?error=not_found`)
  }

  // Fetch account IDs from GoCardless
  let accountIds: string[] = []
  try {
    const requisition = await getRequisition(connection.requisition_id)
    accountIds = requisition.accounts ?? []
  } catch (err) {
    console.error('[banking/callback] failed to get requisition:', err)
    return NextResponse.redirect(`${SITE_URL}/dashboard/banking?error=requisition_failed`)
  }

  // Update connection to linked status with account IDs
  await adminSupabase
    .from('bank_connections')
    .update({ status: 'linked', account_ids: accountIds, updated_at: new Date().toISOString() })
    .eq('id', connection.id)

  return NextResponse.redirect(`${SITE_URL}/dashboard/banking?connected=true`)
}
```

```typescript
// apps/web/lib/banking/syncTransactions.ts
import { SupabaseClient } from '@supabase/supabase-js'
import { getAccountTransactions, GCTransaction } from './goCardlessClient'
import type { TransactionInsert } from '@/types/database'

export function mapGCTransactionToInsert(
  t: GCTransaction,
  userId: string,
  bankConnectionId: string,
): TransactionInsert {
  return {
    user_id: userId,
    amount: Math.abs(parseFloat(t.amount)).toFixed(2),
    merchant: t.merchant,
    date: t.bookingDate,
    source: 'bank_sync',
    external_id: t.externalId,
    bank_connection_id: bankConnectionId,
    note: null,
    category_id: null,
  }
}

export function deduplicateTransactions(
  incoming: GCTransaction[],
  existingExternalIds: Set<string>,
): GCTransaction[] {
  return incoming.filter((t) => t.externalId && !existingExternalIds.has(t.externalId))
}

/**
 * Sync all accounts for a bank connection.
 * Returns the count of new transactions inserted.
 */
export async function syncBankConnection(
  supabase: SupabaseClient,
  connectionId: string,
  userId: string,
): Promise<{ inserted: number; error?: string }> {
  // Fetch the connection
  const { data: connection, error: connError } = await supabase
    .from('bank_connections')
    .select('*')
    .eq('id', connectionId)
    .eq('user_id', userId)
    .single()

  if (connError || !connection) return { inserted: 0, error: 'Connection not found.' }
  if (connection.status !== 'linked') return { inserted: 0, error: 'Connection not linked.' }
  if (!connection.account_ids?.length) return { inserted: 0, error: 'No accounts linked.' }

  // Fetch existing external_ids to avoid re-inserting
  const { data: existing } = await supabase
    .from('transactions')
    .select('external_id')
    .eq('user_id', userId)
    .not('external_id', 'is', null)

  const existingIds = new Set<string>(
    (existing ?? []).map((r: { external_id: string }) => r.external_id)
  )

  // Sync date: 90 days back or since last sync
  const dateFrom = connection.last_synced_at
    ? new Date(connection.last_synced_at).toISOString().split('T')[0]
    : new Date(Date.now() - 90 * 24 * 60 * 60 * 1000).toISOString().split('T')[0]

  let totalInserted = 0

  for (const accountId of connection.account_ids) {
    let transactions: GCTransaction[]
    try {
      transactions = await getAccountTransactions(accountId, dateFrom)
    } catch (err) {
      console.error('[sync] failed to fetch account', accountId, err)
      continue
    }

    const newTransactions = deduplicateTransactions(transactions, existingIds)
    if (newTransactions.length === 0) continue

    const inserts = newTransactions.map((t) =>
      mapGCTransactionToInsert(t, userId, connectionId)
    )

    const { error: insertError } = await supabase
      .from('transactions')
      .insert(inserts)

    if (insertError) {
      console.error('[sync] insert error:', insertError.message)
    } else {
      totalInserted += inserts.length
      inserts.forEach((i) => existingIds.add(i.external_id!))
    }
  }

  // Update last_synced_at
  await supabase
    .from('bank_connections')
    .update({ last_synced_at: new Date().toISOString(), updated_at: new Date().toISOString() })
    .eq('id', connectionId)

  return { inserted: totalInserted }
}
```

```typescript
// apps/web/app/api/banking/sync/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@supabase/supabase-js'
import { syncBankConnection } from '@/lib/banking/syncTransactions'

// This endpoint can be called by:
// - A Supabase pg_cron job (every 6 hours)
// - A GitHub Actions cron workflow
// - Any external scheduler
// Auth: Bearer token must equal CRON_SECRET env var

export async function POST(req: NextRequest) {
  const authHeader = req.headers.get('authorization')
  const cronSecret = process.env.CRON_SECRET
  if (!cronSecret || authHeader !== `Bearer ${cronSecret}`) {
    return NextResponse.json({ error: 'UNAUTHORIZED' }, { status: 401 })
  }

  const adminSupabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_ROLE_KEY!,
    { auth: { autoRefreshToken: false, persistSession: false } }
  )

  const { data: connections } = await adminSupabase
    .from('bank_connections')
    .select('id, user_id')
    .eq('status', 'linked')

  if (!connections?.length) return NextResponse.json({ synced: 0 })

  let totalInserted = 0
  for (const conn of connections) {
    const { inserted } = await syncBankConnection(adminSupabase, conn.id, conn.user_id)
    totalInserted += inserted
  }

  return NextResponse.json({ synced: connections.length, inserted: totalInserted })
}
```

```typescript
import { useEffect } from 'react'
import * as Notifications from 'expo-notifications'
import { Platform } from 'react-native'
import { supabase } from '@/lib/supabase'

Notifications.setNotificationHandler({
  handleNotification: async () => ({
    shouldShowAlert: true,
    shouldPlaySound: true,
    shouldSetBadge: false,
    shouldShowBanner: true,
    shouldShowList: true,
  }),
})

export function usePushNotifications() {
  useEffect(() => {
    registerForPushNotifications()
  }, [])
}

async function registerForPushNotifications() {
  // Physical device only
  if (Platform.OS === 'web') return

  const { status: existingStatus } = await Notifications.getPermissionsAsync()
  let finalStatus = existingStatus

  if (existingStatus !== 'granted') {
    const { status } = await Notifications.requestPermissionsAsync()
    finalStatus = status
  }

  if (finalStatus !== 'granted') return

  try {
    const tokenData = await Notifications.getExpoPushTokenAsync()
    const token = tokenData.data

    // Save token to Supabase profiles table
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) return

    await supabase
      .from('profiles')
      .update({ push_token: token })
      .eq('id', user.id)
  } catch (err) {
    console.error('[push] failed to register push token:', err)
  }
}
```

```typescript
const EXPO_PUSH_URL = 'https://exp.host/--/api/v2/push/send'

export async function sendPushNotification(
  token: string,
  payload: { title: string; body: string; data?: Record<string, unknown> },
): Promise<void> {
  const res = await fetch(EXPO_PUSH_URL, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      Accept: 'application/json',
    },
    body: JSON.stringify({
      to: token,
      title: payload.title,
      body: payload.body,
      data: payload.data ?? {},
      sound: 'default',
    }),
  })

  if (!res.ok) {
    throw new Error(`Expo push failed: ${res.status}`)
  }

  const result = await res.json()
  if (result?.data?.status === 'error') {
    throw new Error(`Expo push error: ${result.data.message}`)
  }
}
```

```typescript
// apps/web/lib/banking/reconcile.ts
import { SupabaseClient } from '@supabase/supabase-js'

export interface ReconciliationCandidate {
  id: string
  amount: string
  date: string
  source: string
  external_id: string | null
}

export interface ReconciliationMatch {
  ocrId: string
  bankId: string
}

const DATE_WINDOW_DAYS = 2
const AMOUNT_TOLERANCE = 0.01

export function findReconciliationMatches(
  ocrTransactions: ReconciliationCandidate[],
  bankTransactions: ReconciliationCandidate[],
): ReconciliationMatch[] {
  const matches: ReconciliationMatch[] = []
  const usedBankIds = new Set<string>()

  for (const ocr of ocrTransactions) {
    const ocrAmount = parseFloat(ocr.amount)
    const ocrDate = new Date(ocr.date).getTime()

    for (const bank of bankTransactions) {
      if (usedBankIds.has(bank.id)) continue

      const bankAmount = parseFloat(bank.amount)
      const bankDate = new Date(bank.date).getTime()
      const daysDiff = Math.abs(ocrDate - bankDate) / (1000 * 60 * 60 * 24)

      if (
        Math.abs(ocrAmount - bankAmount) <= AMOUNT_TOLERANCE &&
        daysDiff <= DATE_WINDOW_DAYS
      ) {
        matches.push({ ocrId: ocr.id, bankId: bank.id })
        usedBankIds.add(bank.id)
        break
      }
    }
  }

  return matches
}

/**
 * Run reconciliation for a user after a bank sync.
 * Updates OCR transactions with the matching bank transaction's external_id as a note annotation.
 */
export async function reconcileForUser(
  supabase: SupabaseClient,
  userId: string,
): Promise<number> {
  const { data: ocrTxns } = await supabase
    .from('transactions')
    .select('id, amount, date, source, external_id')
    .eq('user_id', userId)
    .eq('source', 'ocr')
    .is('external_id', null)

  const { data: bankTxns } = await supabase
    .from('transactions')
    .select('id, amount, date, source, external_id')
    .eq('user_id', userId)
    .eq('source', 'bank_sync')
    .not('external_id', 'is', null)

  if (!ocrTxns?.length || !bankTxns?.length) return 0

  const matches = findReconciliationMatches(
    ocrTxns as ReconciliationCandidate[],
    bankTxns as ReconciliationCandidate[],
  )

  for (const match of matches) {
    const bank = bankTxns.find((t) => t.id === match.bankId)
    if (!bank) continue
    await supabase
      .from('transactions')
      .update({ note: `Matched bank transaction: ${bank.external_id}` })
      .eq('id', match.ocrId)
      .eq('user_id', userId)
  }

  return matches.length
}
```
